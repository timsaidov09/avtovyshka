function fmtTime(v){
  if(!v&&v!==0)return'';
  if(v instanceof Date){
    var h=v.getHours(),m=v.getMinutes();
    return(h<10?'0':'')+h+':'+(m<10?'0':'')+m;
  }
  if(typeof v==='number'&&v>0&&v<1){
    var tm=Math.round(v*24*60);
    var hh=Math.floor(tm/60),mm=tm%60;
    return(hh<10?'0':'')+hh+':'+(mm<10?'0':'')+mm;
  }
  return String(v);
}
function fmtDate(v){
  if(!v)return'';
  if(v instanceof Date){
    var d=v.getDate(),m=v.getMonth()+1,y=v.getFullYear();
    return(d<10?'0':'')+d+'.'+(m<10?'0':'')+m+'.'+y;
  }
  return String(v);
}
